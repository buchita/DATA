SELECT * FROM DT2NN3_B2.CLASS_TABLE;
SELECT * FROM DT2NN3_B2.MODULE_TABLE;
------------------------------------------------
--LECTURER INSERTS CRN AND NUMBER OF STUDENTS IN MODULE
DECLARE
    v_CRN DT2NN3_B2.module_TABLE.crn%TYPE:='&CRN';
    v_NUM_OF_STUDENTS DT2NN3_B2.MODULE_TABLE.NUM_STUDENTS%TYPE:='&NUM_STUDENTS';
    
-- UPDATE THE NUMBER OF STUDENTS IN THE MODULE WITH THE CRN THAT THE LECTURER INSERTED
BEGIN
   UPDATE DT2NN3_B2.MODULE_TABLE
    SET NUM_STUDENTS = v_NUM_OF_STUDENTS
    WHERE CRN = v_CRN; 
    
END;

COMMIT;
